PACKAGE FILES;

IMPORT JAVA.IO.IOEXCEPTION;
IMPORT JAVA.IO.PRINTWRITER;
IMPORT JAVA.NIO.FILE.FILES;
IMPORT JAVA.NIO.FILE.PATHS;
IMPORT JAVA.UTIL.STREAM.STREAM;

/**
 * @AUTHOR LH
 */
PUBLIC CLASS STREAMINANDOUT {
    PUBLIC STATIC VOID MAIN(STRING[] ARGS) {
        TRY (STREAM<STRING> INPUT = FILES.LINES(PATHS.GET("JAVA-BASE", "BOOK", "ON-JAVA", "SRC", "MAIN", "JAVA", "FILES", "STREAMINANDOUT.JAVA"));
             PRINTWRITER OUTPUT = NEW PRINTWRITER("STREAMINANDOUT.TXT")) {
            INPUT.MAP(STRING::TOUPPERCASE).FOREACHORDERED(OUTPUT::PRINTLN);
        } CATCH (IOEXCEPTION E) {
            THROW NEW RUNTIMEEXCEPTION(E);
        }

    }
}
